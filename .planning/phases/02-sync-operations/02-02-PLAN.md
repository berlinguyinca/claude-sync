    **Create src/cli/commands/push.ts:**

    ```typescript
    import type { Command } from "commander";
    import pc from "picocolors";
    import { syncPush } from "../../core/sync-engine.js";
    import type { SyncPushResult } from "../../core/sync-engine.js";
    import { getClaudeDir, getSyncRepoDir } from "../../platform/paths.js";

    export interface PushOptions {
      repoPath?: string;
      claudeDir?: string;
    }

    export async function handlePush(options: PushOptions): Promise<SyncPushResult> {
      return syncPush({
        claudeDir: options.claudeDir ?? getClaudeDir(),
        syncRepoDir: options.repoPath ?? getSyncRepoDir(),
      });
    }

    export function registerPushCommand(program: Command): void {
      program
        .command("push")
        .description("Push local ~/.claude changes to the remote repo")
        .option("--repo-path <path>", "Custom sync repo path", getSyncRepoDir())
        .option("--claude-dir <path>", "Custom ~/.claude path", getClaudeDir())
        .action(async (opts) => {
          try {
            const result = await handlePush(opts);
            if (result.pushed) {
              console.log(pc.green(`Pushed ${result.filesUpdated} files to remote`));
            } else {
              console.log(pc.yellow("No changes to push -- already up to date"));
            }
          } catch (error) {
            const message = error instanceof Error ? error.message : String(error);
            console.error(pc.red(`Push failed: ${message}`));
            process.exitCode = 1;
          }
        });
    }
    ```

    **Create src/cli/commands/pull.ts** -- Same pattern. `handlePull(options)` delegates to `syncPull`. On success, print green message: `"Pulled ${result.filesApplied} files from remote"` and `"Backup saved to: ${result.backupDir}"`. On error, print red `"Pull failed: ${message}"`.

    **Create src/cli/commands/status.ts** -- Same pattern. `handleStatus(options)` delegates to `syncStatus`. Display logic:
    - If `result.isClean && result.remoteDrift.ahead === 0 && result.remoteDrift.behind === 0`: print green "Everything is in sync"
    - If local modifications exist: print header "Local changes:" then each change with type indicator (M for modified, A for added, D for deleted) and the path
    - If `remoteDrift.behind > 0`: print yellow "Remote is ${behind} commit(s) ahead -- run 'claude-sync pull'"
    - If `remoteDrift.ahead > 0`: print "Local is ${ahead} commit(s) ahead -- run 'claude-sync push'"
    - Always print dim "Excluded: ${excludedCount} files (not in sync manifest)"
    - If `!result.hasRemote`: print yellow "No remote configured"
